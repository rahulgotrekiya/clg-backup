declare
   cursor cApp is select aid, a_name from applicant;
   cursor c1(vID applicant.aid%type) is
   select et.et_id, et_name, score from etest_details etd
   join entrance_test et
   on etd.et_id = et.et_id where etd.aid = vID;
   vtot number(5);
   vID applicant.aid%type;
   vName applicant.a_name%type;
begin
   for vCurApp in cApp
   loop
           vtot := 0;
           vID := vCurApp.aid;
           dbms_output.put_line('Applicant Name: ' || vCurApp.a_Name);
           dbms_output.put_line('Test ID' || 'Test Name' || 'Score' );
           for vCur in c1(vID)
           loop
                   dbms_output.put(rpad(vCur.et_id, 5, ' '));
                   dbms_output.put(rpad(vCur.et_name, 20, ' '));
                   dbms_output.put_line(lpad(vCur.score, 5, ' '));
                   vtot := vtot + vCur.score;
           end loop;
           dbms_output.put_line(lpad('-',255, ' '));
           dbms_output.put_line(lpad('Total: ',55,' ') || vtot);
           dbms_output.put_line('*********************************');
   end loop;
exception
   when NO_DATA_FOUND then
           dbms_output.put_line('Applicant data not available');
end;
/
