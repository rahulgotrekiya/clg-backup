CREATE TABLE Employee (
    emp_no NUMBER PRIMARY KEY,
    fname VARCHAR2(50),
    lname VARCHAR2(50),
    dob DATE,
    address VARCHAR2(100)
);


CREATE TABLE Company (
    comp_id NUMBER PRIMARY KEY,
    emp_no NUMBER,
    company_name VARCHAR2(100),
    salary NUMBER,
    department VARCHAR2(50),
    designation VARCHAR2(50),
    FOREIGN KEY (emp_no) REFERENCES Employee(emp_no)
);

CREATE TABLE Customer (
    cust_no NUMBER PRIMARY KEY,
    fname VARCHAR2(50),

1. Find all students whose salary is higher than the average salary of the employees in their departments. 

SELECT e.emp_no, e.fname, c.department, c.salary FROM Employee e
JOIN Company c ON e.emp_no = c.emp_no
WHERE c.salary > (
	SELECT AVG(c2.salary)
	FROM Company c2
	WHERE c2.department = c.department
);
