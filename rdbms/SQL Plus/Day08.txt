2. Display the item which is distributed maximum time

SELECT i.Item_Name, COUNT(di.Item_No) AS Times_Distributed
FROM Dist_Item di
JOIN Item i
ON di.Item_No = i.Item_No
GROUP BY i.Item_Name
HAVING COUNT(di.Item_No) = (
    SELECT MAX(COUNT(di2.Item_No))
    FROM Dist_Item di2
    GROUP BY di2.Item_No
);

SELECT i.Item_Name, di.Qty, di.Date_Supplied
FROM Item i
INNER JOIN Dist_Item di ON i.Item_No = di.Item_No
WHERE EXTRACT(MONTH FROM di.Date_Supplied) = 7
AND EXTRACT(YEAR FROM di.Date_Supplied) = 2021;

1. Display all the workers assigned for Job “Assemble”

SELECT w.Id, w.Name, w.Wages_Per_Hr, a.Start_Date, a.Status, a.Total_Hrs
FROM Worker w
INNER JOIN Assigned a ON w.Id = a.W_Id
INNER JOIN Job j ON a.J_Id = j.Job_Id
WHERE j.Job_Type = 'Assemble';

2. Display total hours of the worker “Deep Jha” spent on Job “Packing”

SELECT w.Name, j.Job_Type, SUM(a.Total_Hrs) AS Total_Hours_Spent
FROM Worker w
INNER JOIN Assigned a ON w.Id = a.W_Id
INNER JOIN Job j ON a.J_Id = j.Job_Id
WHERE w.Name = 'Deep Jha' AND j.Job_Type = 'Packing'
GROUP BY w.Name, j.Job_Type;

3. Display all the job types that are starting from Sept-2021.

SELECT DISTINCT j.Job_Type, a.Start_Date
FROM Job j
INNER JOIN Assigned a ON j.Job_Id = a.J_Id
WHERE EXTRACT(MONTH FROM a.Start_Date) = 9
AND EXTRACT(YEAR FROM a.Start_Date) = 2021
ORDER BY a.Start_Date;

4. What are the total hours allocated for Job “Cutting”

SELECT j.Job_Type, SUM(a.Total_Hrs) AS Total_Hours_Allocated
FROM Job j
INNER JOIN Assigned a ON j.Job_Id = a.J_Id
WHERE j.Job_Type = 'Cutting'
GROUP BY j.Job_Type;

5. Display all the jobs that are still pending

SELECT j.Job_Id, j.Job_Type, w.Name AS Worker_Name, a.Start_Date
FROM Job j
INNER JOIN Assigned a ON j.Job_Id = a.J_Id
INNER JOIN Worker w ON a.W_Id = w.Id
WHERE a.Status = 'Pending'
ORDER BY a.Start_Date;
