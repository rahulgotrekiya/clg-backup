// College menegement system

create table student (
	rollno number(3),
	name varchar(25),
	enrollment number(16),
	gender char(1),
	course varchar(25),	
	contactno number(10),
	address varchar(25)
)
/

create table faculty (
	fid number(3),
	name varchar(25),
	gender char(1),
	subject varchar(25),	
	department varchar(25),	
	contactno number(10),
	address varchar(25)
)
/

create table subject (
	code varchar(3),
	subname varchar(25),
	faculty varchar(50),
	credit number(1),
	hours number(2)
)
/

create table course (
	id number(3),
	name varchar(25),
	department varchar(25),	
	type char(2)
)
/

create table department (
	id number(3),
	name varchar(25),
	totalstudents number(10),
	location varchar(25),
	contact number(10)
)
/
