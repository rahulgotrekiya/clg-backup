declare
	cursor cEmp(vDno emp.deptno%type) is select ename,sal from emp where deptno = vDNo;
	tot number(8,2);
	vDno emp.deptno%type;
begin
	vDno := &DeptNo;
	tot := 0;
	for vCur in cEmp(vDNo)
	loop
		dbms_output.put_line(rpad('Employee ',12,' ') || vCur.ename   || ' is getting ' || vCur.sal  || ' per month');
		tot := tot + vCur.sal;
	end loop;
	dbms_output.put_line(tot);
exception
	when NO_DATA_FOUND then
		dbms_output.put_line('Emp No does not exists..');
end;
/
