set serveroutput on;

-- Odd Even

DECLARE
    n number(3);
BEGIN
    n := &no;
    if MOD(n,2) = 0 then
        dbms_output.put_line(n || ' is Even');
    else
        dbms_output.put_line(n || ' is Odd');
    end if;
    dbms_output.put_line(n);
end;
/

-- Exception and %type

DECLARE
    vName applicant.a_name%type;
    vId applicant.aid%type;
BEGIN
    vId := '&Id';
    select a_name into vName from applicant where aid = vId;
    dbms_output.put_line(vName);
EXCEPTION 
    WHEN NO_DATA_FOUND then
        dbms_output.put_line('Wrong applicant Id !!!');
END;
/

-- Multiple colums

DECLARE
    vId applicant.aid%type;
    vName applicant.a_name%type;
    vCity applicant.city%type;
BEGIN
    vId := '&Id';
    select a_name, city into vName, vCity from applicant where aid = vId;
    dbms_output.put_line(vName || ' Lives in ' || vCity);
EXCEPTION 
    WHEN NO_DATA_FOUND then
        dbms_output.put_line('Wrong applicant Id !!!');
END;
/

-- Loop 
DECLARE
    n NUMBER(2);
BEGIN
    n := &n;
    LOOP
        dbms_output.put_line(n);
        n := n-1;
        exit when n > 0;
    END LOOP;
END;
/

-- while loop
DECLARE
    n NUMBER(2);
BEGIN
    n := &n;
    while n > 0
    LOOP
        dbms_output.put_line(n);
        n := n-1;
    END LOOP;
END;
/

-- For loop
BEGIN
     FOR n in 1 .. 10
    LOOP
        dbms_output.put_line(n);
    END LOOP;
END;
/

-- For loop reverse
BEGIN
    FOR n in reverse 1 .. 10
    LOOP
        dbms_output.put_line(n);
    END LOOP;
END;
/

DECLARE 
    CURSOR curEmp is SELECT empno, ename, sal FROM emp;
    vNo emp.empno%type;
    vName emp.ename%type;
    vSal emp.sal%type;
BEGIN 
    open curEmp;
    LOOP
        FETCH curEmp INTO vNo, vName, vSal;
        dbms_output.put_line('Emp no is: ' || vNo);
        dbms_output.put_line('Emp no is: ' || vName);
        dbms_output.put_line('Emp no is: ' || vSal);
        dbms_output.put_line('------------------');
        exit WHEN curEmp%NOTFOUND;
    END LOOP;
    close curEmp;
END;
/
