    issue_id     VARCHAR2(5) PRIMARY KEY,

    book_code    VARCHAR2(5) REFERENCES BOOK_CATALOG(book_code),
    issue_date   DATE NOT NULL,
    issue_ret_dt DATE CHECK (issue_ret_dt >= issue_date)
);

SELECT Part_no, Part_name 
FROM Participants 
WHERE Part_no IN (
   SELECT Part_no
   FROM Scorecard s
   JOIN Competition c
   ON s.Comp_code = c.Comp_code
   WHERE c.Comp_name = 'Dancing'
)
AND Part_no IN (
   SELECT Part_no FROM Scorecard s
   JOIN Competition c ON s.Comp_code = c.Comp_code
   WHERE c.Comp_name = 'Painting'
);

-- Supplier table
CREATE TABLE SUPPLIER (
    s_id          VARCHAR2(5) PRIMARY KEY,
    s_name        VARCHAR2(50) NOT NULL,
    contact       VARCHAR2(15)
);

-- Parts table
CREATE TABLE PARTS (
    part_id       VARCHAR2(5) PRIMARY KEY,
    part_name     VARCHAR2(50) NOT NULL,
    colour        VARCHAR2(10) CHECK (colour IN ('BLACK','WHITE','GREY')),
    unit_rate     NUMBER(10,2) CHECK (unit_rate >= 0)
);

-- Catalog table
CREATE TABLE CATALOG (
    s_id    VARCHAR2(5) REFERENCES SUPPLIER(s_id),
    part_id VARCHAR2(5) REFERENCES PARTS(part_id),
    qty     NUMBER(5) CHECK (qty >= 0),
    PRIMARY KEY (s_id, part_id)
);
