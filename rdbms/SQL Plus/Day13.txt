1.	Write a procedure which will take Faculty ID as an input and will display all the information of that faculty 

CREATE or REPLACE procedure p1 
(vid in faculty.fid%type)
is
    vname faculty.name%type;
    vgender Faculty.gender%type;
    vsubject Faculty.subject%type;
    vdept Faculty.department%type;
    vcontact Faculty.contactno%type;
    varea Faculty.area%type;
    vexp Faculty.experience%type;
BEGIN
    select name, gender, subject, department, contactno, area, experience into vname, vgender, vsubject, vdept, vcontact, varea, vexp from faculty where fid = vid;

    dbms_output.put_line(vname || ' ' || vgender || ' ' || vsubject || ' ' || vdept || ' ' || vcontact || ' ' || varea || ' ' || vexp);
END;
/

3.	Write a procedure which will take minimum limit and maximum limit of salary and the 
execution of the procedure will display name of the employees having salary between the 
range. 

CREATE OR REPLACE PROCEDURE show_employees_in_salary_range(
    p_min IN NUMBER,
    p_max IN NUMBER
)
IS
BEGIN
    FOR emp IN (
        SELECT Emp_Name, Basic_Salary
        FROM Employee_master
        WHERE Basic_Salary BETWEEN p_min AND p_max
    )
    LOOP
        DBMS_OUTPUT.PUT_LINE(emp.Emp_Name || ' — ' || emp.Basic_Salary);
    END LOOP;
END;
/

1. Write a PLSQL block to display total number of products ordered in a user entered Order_ID 

DECLARE
    v_order_id NUMBER := &OrderID;
    v_total NUMBER;
BEGIN
    SELECT SUM(Quantity)
    INTO v_total
    FROM ORDER_DETAILS
    WHERE Order_ID = v_order_id;

    DBMS_OUTPUT.PUT_LINE('Total products ordered = ' || v_total);
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('No such Order_ID');
END;
/

DECLARE

5. Write a PLSQL block to take Supplier_Id as input and display total number of products they supply 

DECLARE
    v_sup NUMBER := &Supplier_ID;
    v_total NUMBER;
BEGIN
    SELECT COUNT(Prod_ID)
    INTO v_total
    FROM PRODUCTS
    WHERE Supplier_ID = v_sup;

    DBMS_OUTPUT.PUT_LINE('Supplier ' || v_sup || ' supplies ' || v_total || ' products.');
END;
